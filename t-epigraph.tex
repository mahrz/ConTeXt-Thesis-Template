\def\Epigraph[#1][#2][#3]
{
  \rbox {
      \startepi
        #1
      \stopepi
      \vskip0.1cm
      \cap{#2}, #3
  }
  \blank[5*medium]
}

{
  \rbox {\tfx
      \startepi
        #1
      \stopepi
      \vskip0.1cm
      \cap{#2}, #3
  }
  \blank[3*medium]
}
