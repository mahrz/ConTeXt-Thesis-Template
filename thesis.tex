% Overall project
\project project_thesis

% Thesis specific layout
\environment env_thesis_layout
\setupnotation[footnote][way=bypage,numbercommand=\tf]

% Use pdf links & clickable urls
\setupinteraction[state=start]

% Use this module to print nice epigraphs
\usemodule[epigraph]

\version[concept]
\language[en-gb]

%\showlayout

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Title
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% The title page
% We want a pretitle
\enablemode[pretitled]
\component title

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Front Matter
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\startfrontmatter
% Counting starts after the title
\setuppagenumber[number=1]
% Let the page numbering appear with roman numerals
% in the middle
\setuppagenumbering[location=middle]
\setuppagenumber[numberconversion=romannumerals]

% Compact mode removes forcing frontmatter components
% to start on an odd page
% \enablemode[compact]

\component abstract
\component dedication
\component epigraph


\stopfrontmatter

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Body Matter
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\startbodymatter
% Let the page numbering reappear
\setuppagenumber[numberconversion=numbers]
\setuppagenumbering[conversion=numbers,
                    location=right]

% Footnotes
\setupfootnotes[rule=off]

% One and half line spacing (if needed)
\setupinterlinespace[medium]

\component intro
\component pics
\component math
\stopbodymatter
