% Typeset the title page, including the backpage
\def\TitlePage
{
  \setuppagenumbering[location=]
  \setuplayout[titles]
  \startalignment[center]
    \tfd {\bf \useblocks[title][title]}
    \blank[3*medium]
    \tfc {\useblocks[title][subtitle]}
    \blank[4*medium]
    \tfb \useblocks[title][author]
    \blank[2*medium]
    \tfa \useblocks[title][date]
  \stopalignment
  \page[empty]
  \setuplayout[reset]
}
